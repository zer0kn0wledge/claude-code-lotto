Fisher-Yates Shuffle

  seededShuffle(array, seedHex):
    rng = createPRNG(seedHex)
    a = copy(array)
    for i from (a.length - 1) down to 1:
      j = floor(rng.next() * (i + 1))
      swap(a[i], a[j])
    return a

Draw Procedure

  1. serverSeed = random secret string
  2. seedHash = SHA256(serverSeed)       ← published BEFORE draw
  3. shuffleKey = SHA256(serverSeed + ":" + entrantCount)
  4. shuffled = seededShuffle(entrants, shuffleKey)
  5. winners = shuffled[0], shuffled[1], shuffled[2]
  6. serverSeed revealed AFTER draw      ← anyone can now verify

import { createHash } from 'crypto';

function sha256(s) {
  return createHash('sha256').update(s).digest('hex');
}

function createPRNG(seedHex) {
  let counter = 0;
  return () => {
    const hash = sha256(seedHex + ':' + counter++);
    return parseInt(hash.slice(0, 8), 16) / 0x100000000;
  };
}

function seededShuffle(arr, seedHex) {
  const rng = createPRNG(seedHex);
  const a = [...arr];
  for (let i = a.length - 1; i >= 1; i--) {
    const j = Math.floor(rng() * (i + 1));
    [a[i], a[j]] = [a[j], a[i]];
  }
  return a;
}

import hashlib

def sha256(s):
    return hashlib.sha256(s.encode()).hexdigest()

def create_prng(seed_hex):
    counter = [0]
    def next_val():
        h = sha256(f"{seed_hex}:{counter[0]}")
        counter[0] += 1
        return int(h[:8], 16) / 0x100000000
    return next_val

def seeded_shuffle(arr, seed_hex):
    rng = create_prng(seed_hex)
    a = list(arr)
    for i in range(len(a) - 1, 0, -1):
        j = int(rng() * (i + 1))
        a[i], a[j] = a[j], a[i]
    return a
